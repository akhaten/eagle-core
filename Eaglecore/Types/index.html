class snr_db(float):
    """
    Class to represent Signal to Noise ratio in dB.
    """

    def __str__(self) -> str:
        return '{} [dB]'.format(super().__str__())

    def to_linear(self) -> 'snr':
        """Convert to SNR in linear.

        Returns:
            linear SNR
        """
        return snr(10 ** (self / 10))

Formula
\[\begin{align} SNR_{dB} &= 10 \times log_{10}(SNR) \\ \frac{SNR_{dB}}{10} &= log_{10}(SNR) \\ SNR &= 10^{\frac{SNR_{dB}}{10}} \end{align}\]
\begin{align}
SNR_{dB} &= 10 \times log_{10}(SNR) \\
\frac{SNR_{dB}}{10} &= log_{10}(SNR) \\
SNR &= 10^{\frac{SNR_{dB}}{10}}
\end{align}

Do not confusion with formula using amplitudes. See wiki.
